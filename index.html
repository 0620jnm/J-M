uploadBtn.onclick = async () => { if (!selectedFiles.length) { alert("請先選擇照片"); return; } uploadBtn.disabled = true; const progressContainer = document.getElementById("progressContainer"); const progressBar = document.getElementById("progressBar"); const uploadStatus = document.getElementById("uploadStatus"); progressContainer.style.display = "block"; progressBar.style.width = "0%"; let successCount = 0; let uniqueErrors = new Set(); // 用來儲存不重複的失敗原因 for (let i = 0; i < selectedFiles.length; i++) { uploadStatus.innerHTML = `正在上傳 (${i + 1} / ${selectedFiles.length})...`; try { const formData = new FormData(); formData.append("file", selectedFiles[i]); formData.append("upload_preset", UPLOAD_PRESET); const cloudRes = await fetch(`https://api.cloudinary.com/v1_1/${CLOUD_NAME}/image/upload`, { method: "POST", body: formData }); const cloudData = await cloudRes.json(); if (!cloudRes.ok) { throw new Error(cloudData.error.message || "連線不穩定"); } await addDoc(photosCol, { url: cloudData.secure_url, ownerId: currentUser.uid, createdAt: serverTimestamp() }); successCount++; } catch (err) { console.error("Upload Error:", err); // 根據常見錯誤簡化訊息 let msg = err.message; if (msg.includes("exceeds the limit")) msg = "照片檔案太大"; if (msg.includes("Failed to fetch")) msg = "網路連線中斷"; uniqueErrors.add(msg); } progressBar.style.width = `${((i + 1) / selectedFiles.length) * 100}%`; } // --- 最終結果顯示 --- if (successCount === selectedFiles.length) { uploadStatus.innerHTML = "✅ 全部上傳成功！✨"; } else if (successCount === 0) { uploadStatus.innerHTML = `❌ 上傳失敗，原因為：${Array.from(uniqueErrors).join("、")}`; } else { uploadStatus.innerHTML = `⚠️ 部分上傳失敗，失敗原因為：${Array.from(uniqueErrors).join("、")}`; } // 重置選取 selectedFiles = []; renderPreviews(); setTimeout(() => { uploadBtn.disabled = false; // 如果全成功才自動隱藏進度條，有錯的話留著讓使用者看原因 if (successCount === (successCount + uniqueErrors.size)) { progressContainer.style.display = "none"; uploadStatus.textContent = ""; } }, 6000); await loadGallery(false); };
